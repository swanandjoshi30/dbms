CREATE DATABASE assignment5;

use assignment5;

CREATE TABLE IF NOT EXISTS areas (
    radius DECIMAL(5, 2),
    area DECIMAL(10, 2)
);


DELIMITER //
CREATE PROCEDURE calculate_area()
BEGIN
    DECLARE r INT DEFAULT 5;
    DECLARE pi DECIMAL(5,2) DEFAULT 3.14;
    DECLARE area DECIMAL(10,2);
    
    WHILE r <= 9 DO
        SET area = pi * r * r;
        INSERT INTO areas (radius, area) VALUES (r, area);
        SET r = r + 1;
    END WHILE;
END //

DELIMITER ;
CALL calculate_area();

SELECT * FROM areas;
